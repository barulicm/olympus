set(CPACK_PACKAGE_NAME "Olympus")
set(CPACK_PACKAGE_VENDOR "Matthew Barulic")
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "Unofficial score keeping software for FIRST LEGO League")
set(CPACK_PACKAGE_INSTALL_DIRECTORY ${CPACK_PACKAGE_NAME})
set(CPACK_PACKAGE_VERSION_MAJOR ${PROJECT_VERSION_MAJOR})
set(CPACK_PACKAGE_VERSION_MINOR ${PROJECT_VERSION_MINOR})
set(CPACK_PACKAGE_VERSION_PATCH ${PROJECT_VERSION_PATCH})
set(CPACK_VERBATIM_VARIABLES YES)
set(CPACK_PACKAGE_DESCRIPTION_FILE ${CMAKE_CURRENT_LIST_DIR}/Description.txt)
set(CPACK_RESOURCE_FILE_WELCOME ${CMAKE_CURRENT_LIST_DIR}/Welcome.txt)
set(CPACK_RESOURCE_FILE_LICENSE ${CMAKE_CURRENT_LIST_DIR}/License.txt)
set(CPACK_RESOURCE_FILE_README ${CMAKE_CURRENT_LIST_DIR}/Readme.txt)
set(CPACK_CREATE_DESKTOP_LINKS olympus_executable)
set(CPACK_PACKAGE_EXECUTABLES "Olympus" "Olympus" )

if(WIN32)
    set(CPACK_GENERATOR NSIS)
elseif(APPLE)
    set(CPACK_GENERATOR DMG)
elseif(CMAKE_SYSTEM_NAME STREQUAL "Linux")
    set(CPACK_GENERATOR DEB)
else()
    set(CPACK_GENERATOR TGZ)
endif()

# NSIS generator config
set(CPACK_NSIS_MUI_ICON ${CMAKE_CURRENT_LIST_DIR}/Olympus.ico)
set(CPACK_NSIS_MUI_UNIICON ${CMAKE_CURRENT_LIST_DIR}/Olympus.ico)
set(CPACK_NSIS_INSTALLED_ICON_NAME bin/Olympus.exe)
set(CPACK_NSIS_DISPLAY_NAME "Olympus Scoring Software")
set(CPACK_NSIS_PACKAGE_NAME "Olympus")
set(CPACK_NSIS_ENABLE_UNINSTALL_BEFORE_INSTALL YES)

# DEB generator config
set(CPACK_DEBIAN_PACKAGE_MAINTAINER "Matthew Barulic")
set(CPACK_DEBIAN_PACKAGE_DEPENDS "libcpprest2.10")

include(CPack)
